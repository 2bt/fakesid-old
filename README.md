    rausch              # the project folder; its name doesn't matter
    SDL2-2.0.7          # SDL source
    SDL2_image-2.0.2    # SDL image source
    SDL2_mixer-2.0.2    # SDL mixer source
    tools_r25.2.5-linux # android SDK
